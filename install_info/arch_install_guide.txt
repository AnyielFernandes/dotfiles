loadkeys es
timedatectl set-ntp true #timedatectl status 

#Partitioning the disk
parted /dev/sda
mklabel gpt (probablemente no habrá que ejecutar en software real)

#En máquina virtual: 
mklabel gpt 
mkpart primary 0% 260MiB
set 1 boot esp
mkpart primary 273MB 800MB 
set 2 swap

#Formatear sistemas de ficheros
mkfs.ext4 /dev/sda3
mkfs.fat -F32 /dev/sda1
mkswap /dev/sda2
swapon /dev/sda2

#Montar sistemas de ficheros
mount /dev/sda3 /mnt

#Instalación: 
pacstrap /mnt base linux linux-firmware

genfstab -U /mnt >> /mnt/etc/fstab
arch-chroot /mnt
ln -sf /usr/share/zoneinfo/Europe/Madrid /etc/localtime
hwclock --systohc

pacman -S vi 
vi /etc/locale.gen
#Descomentar es_ES.UTF-8 UTF-8
locale-gen

#Hostname
/etc/hostname #escribir hostname
#Configurar /etc/hosts
127.0.0.1   localhost
::1         localhost
127.0.1.1   Hostname.localdomain    hostname


crear usuarios: 

useradd -m anyel 

usermod -aG wheel,audio,video,optical,storage anyel 
pacman -S sudo 
EDITOR="vi" visudo
#Descomentar linea de wheel que permite que todos los usuarios del grupo puedan ejecutar cualquier comando.


#Instalar grub
pacman -S grub
pacman -S efibootmgr dosfstools os-prober mtools 
mkdir /boot/EFI
mount /dev/sda1 /boot/EFI
grub-install --target=x86_64-efi --bootloader-id=grub_uefi --recheck 
grub-mkconfig -o /boot/grub/grub.cfg


pacman -S networkmanager git
grub-mkconfig -o /boot/grub/grub.cfg


pacman -S networkmanager gi
systemctl enable NetworkManager.service

exit 
umount -l /mnt
